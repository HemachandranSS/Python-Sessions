üìã Procedural Programming

A step-by-step approach where code is organized as a sequence of functions that operate on data.


    # Procedural approach to calculate area
    
    def get_rectangle_area(length, width):
        """Calculate the area of a rectangle."""
        return length * width
    
    
    def get_circle_area(radius):
        """Calculate the area of a circle."""
        return 3.14159 * radius ** 2
    
    
    def display_area(shape, area):
        """Display the area of a shape."""
        print(f"{shape} area: {area:.2f}")
    
    
    # Main execution
    if __name__ == "__main__":
        rect_area = get_rectangle_area(5, 3)
        circle_area = get_circle_area(4)
    
        display_area("Rectangle", rect_area)
        display_area("Circle", circle_area)

üèóÔ∏è Object-Oriented Programming

Organizes code into classes and objects, encapsulating data and methods together.


    # Object-oriented approach
    from abc import ABC, abstractmethod
    
    
    class Shape(ABC):
        """Abstract base class for geometric shapes."""
    
        def __init__(self, name):
            self.name = name
            self.area = 0
    
        @abstractmethod
        def calculate_area(self):
            """Calculate the area of the shape."""
            pass
    
        def display_area(self):
            """Display the calculated area."""
            self.calculate_area()
            print(f"{self.name} area: {self.area:.2f}")
    
    
    class Rectangle(Shape):
        """Rectangle class inheriting from Shape."""
    
        def __init__(self, length, width):
            super().__init__("Rectangle")
            self.length = length
            self.width = width
    
        def calculate_area(self):
            """Calculate rectangle area."""
            self.area = self.length * self.width
    
    
    class Circle(Shape):
        """Circle class inheriting from Shape."""
    
        def __init__(self, radius):
            super().__init__("Circle")
            self.radius = radius
    
        def calculate_area(self):
            """Calculate circle area."""
            self.area = 3.14159 * self.radius ** 2
    
    
    # Main execution
    if __name__ == "__main__":
        rect = Rectangle(5, 3)
        circle = Circle(4)
    
        rect.display_area()
        circle.display_area()

üß© Modular Programming

Breaking code into separate, reusable modules for better organization and maintainability.


    # File: math_operations.py
    """Module containing basic mathematical operations."""

    def add(a, b):
        """Add two numbers together."""
        return a + b


    def multiply(a, b):
        """Multiply two numbers together."""
        return a * b


    def power(base, exponent):
        """Calculate base raised to the power of exponent."""
        return base ** exponent


    # File: constants.py
    """Mathematical constants."""

    PI = 3.14159265359
    E = 2.71828182846


    # File: geometry.py
    """Module for geometric calculations."""

    from math_operations import multiply, power
    from constants import PI


    def rectangle_area(length, width):
        """Calculate the area of a rectangle."""
        return multiply(length, width)


    def circle_area(radius):
        """Calculate the area of a circle."""
        return multiply(PI, power(radius, 2))


    def triangle_area(base, height):
        """Calculate the area of a triangle."""
        return multiply(multiply(base, height), 0.5)


    # File: main.py
    """Main application using modular components."""

    from geometry import rectangle_area, circle_area, triangle_area
    from math_operations import add


    def main():
        """Main function to demonstrate modular programming."""
        # Calculate individual areas
        rect_area = rectangle_area(5, 3)
        circ_area = circle_area(4)
        tri_area = triangle_area(6, 8)

        # Calculate total area
        total_area = add(add(rect_area, circ_area), tri_area)

        # Display results
        print(f"Rectangle area: {rect_area:.2f}")
        print(f"Circle area: {circ_area:.2f}")
        print(f"Triangle area: {tri_area:.2f}")
        print(f"Total area: {total_area:.2f}")


    if __name__ == "__main__":
        main()
