#!/usr/bin/env python3
"""
Cross-platform file operations demo.

This module demonstrates Python's platform independence
by showing system info and file operations that work
seamlessly across Windows, macOS, and Linux.
"""

import os
import platform
import sys
from pathlib import Path
from typing import List, Tuple


def get_system_info() -> dict:
    """
    Get comprehensive system information.
    
    Returns:
        dict: Dictionary containing system details
    """
    return {
        'system': platform.system(),
        'platform': platform.platform(),
        'architecture': platform.architecture()[0],
        'python_version': platform.python_version(),
        'machine': platform.machine(),
        'processor': platform.processor() or 'Unknown'
    }


def display_system_info() -> None:
    """Display formatted system information."""
    info = get_system_info()
    
    print("üñ•Ô∏è  SYSTEM INFORMATION")
    print("=" * 30)
    
    for key, value in info.items():
        formatted_key = key.replace('_', ' ').title()
        print(f"{formatted_key:<15}: {value}")


def create_cross_platform_path(base_dir: str, filename: str) -> Path:
    """
    Create platform-independent file paths using pathlib.
    
    Args:
        base_dir: Base directory path
        filename: Name of the file
        
    Returns:
        Path: Platform-independent path object
    """
    return Path(base_dir) / filename


def get_directory_info(directory: str = ".") -> Tuple[List[str], int, int]:
    """
    Get information about files and directories.
    
    Args:
        directory: Directory to analyze (default: current directory)
        
    Returns:
        Tuple containing (files_list, file_count, dir_count)
    """
    try:
        path = Path(directory)
        items = list(path.iterdir())
        
        files = [item.name for item in items if item.is_file()]
        dirs = [item.name for item in items if item.is_dir()]
        
        return files, len(files), len(dirs)
    
    except (PermissionError, FileNotFoundError) as e:
        print(f"Error accessing directory: {e}")
        return [], 0, 0


def demonstrate_file_operations() -> None:
    """Demonstrate cross-platform file operations."""
    print("\nüìÅ FILE OPERATIONS DEMO")
    print("=" * 30)
    
    # Get user's home directory (works on all platforms)
    home_dir = Path.home()
    print(f"Home Directory    : {home_dir}")
    
    # Create cross-platform file path
    demo_file = create_cross_platform_path(home_dir, "python_demo.txt")
    print(f"Demo File Path    : {demo_file}")
    
    # Get current working directory
    current_dir = Path.cwd()
    print(f"Current Directory : {current_dir}")
    
    # Analyze current directory
    files, file_count, dir_count = get_directory_info()
    print(f"Files Found       : {file_count}")
    print(f"Directories Found : {dir_count}")
    
    # Display some files (max 5)
    if files:
        print(f"Sample Files      : {', '.join(files[:5])}")
        if len(files) > 5:
            print(f"                    ... and {len(files) - 5} more")


def demonstrate_platform_features() -> None:
    """Show platform-specific features and differences."""
    print("\nüåç PLATFORM FEATURES")
    print("=" * 30)
    
    system = platform.system().lower()
    
    # Path separator demonstration
    print(f"Path Separator    : '{os.sep}'")
    print(f"Line Separator    : {repr(os.linesep)}")
    
    # Platform-specific information
    if system == 'windows':
        print(f"Windows Edition   : {platform.win32_edition()}")
        print(f"Windows Version   : {platform.win32_ver()[0]}")
    elif system == 'darwin':  # macOS
        print(f"macOS Version     : {platform.mac_ver()[0]}")
    elif system == 'linux':
        distro = platform.freedesktop_os_release()
        print(f"Linux Distribution: {distro.get('NAME', 'Unknown')}")
        print(f"Version           : {distro.get('VERSION', 'Unknown')}")


def main() -> None:
    """Main function to run all demonstrations."""
    print("üêç PYTHON PLATFORM INDEPENDENCE DEMO")
    print("=" * 50)
    print("This same code runs identically on Windows, macOS, and Linux!")
    print("=" * 50)
    
    # Run all demonstrations
    display_system_info()
    demonstrate_file_operations()
    demonstrate_platform_features()
    
    print("\n‚úÖ Demo completed successfully!")
    print("üí° The same Python bytecode ran on your platform!")


if __name__ == "__main__":
    main()
