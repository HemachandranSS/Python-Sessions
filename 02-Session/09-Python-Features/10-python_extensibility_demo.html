C Extension with ctypes

                

import ctypes

# Load math library
libc = ctypes.CDLL('libc.so.6')
libc.pow.argtypes = (ctypes.c_double, ctypes.c_double)
libc.pow.restype = ctypes.c_double

# Use C function
result = libc.pow(2.0, 10.0)
print(f"C pow(2.0, 10.0) = {result}")

# Pure Python vs C Extension
def fibonacci_python(n):
    if n <= 1: return n
    return fibonacci_python(n-1) + fibonacci_python(n-2)

# C extension would be ~100x faster
import time
start = time.time()
result = fibonacci_python(35)
duration = time.time() - start
